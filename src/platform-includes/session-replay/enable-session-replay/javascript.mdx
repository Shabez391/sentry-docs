Install the session replay package:

```bash {tabTitle: ESM}
# Using yarn
yarn add @sentry/replay

# Using npm
npm install --save @sentry/replay
```

```html {tabTitle: CDN}
<script
  <!--
    Note that `bundle.replay.min.js` contains both `@sentry/browser` AND
    `@sentry/replay`, and should therefore be used in place of
    `@sentry/browser`'s bundle rather than in addition to it.
  -->
  src="https://browser.sentry-cdn.com/{{ packages.version('sentry.javascript.browser') }}/bundle.replay.min.js"
  integrity="sha384-{{ packages.checksum('sentry.javascript.browser', 'bundle.replay.min.js', 'sha384-base64') }}"
  crossorigin="anonymous"
></script>
```
